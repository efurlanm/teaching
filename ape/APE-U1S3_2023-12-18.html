Operadores aritmticos de incremento e decremento

In[21]:
#include <stdio.h>
int main(){
    int a, b;
    a = 5;
    b = ++a;
    printf("%d, %d\n", a, b);

    a = 5;
    b = --a;
    printf("%d, %d\n", a, b);
    
    a = 5;
    b = a++;
    printf("%d, %d\n", a, b);

    a = 5;
    b = a--;
    printf("%d, %d\n", a, b);
}

6, 6
4, 4
6, 5
4, 5
Operador Operao Retorna o valor
++a incrementa a varivel incrementado
--a decrementa a varivel decrementado
a++ incrementa a varivel original antes da operao de incremento
a-- decrementa a varivel original antes da operao de decremento
