Filas circulares

So os ponteiros, e no os elementos da fila, que se movem em direo ao incio do vetor

In[]:
/* Vamos definir a constante N com valor de 10 */

#define N 10

struct filacirc {
    /* Criao da es trutura da Fila Circular */
    int tam, ini, fim;

    char vet[N];
};

typedef struct filacirc FilaCirc;

/* Funo para inicializar a Fila */
void inicia_fila(FilaCirc * f) {
    f -> tam = 0;
    f -> ini = 1;
    f -> fim = 0;
}

/* Funo para inserir na Fila */
void insere_fila(FilaCirc * f, char elem) {
    if (f -> tam == N - 1) {
        /* Verifica se a Fila est completa */
        printf(A fila esta cheia\ n);
    } else {
        /* Caso a Fila no esteja completa, inserimos o elemento */
        f -> fim = (f -> fim % (N - 1)) + 1;
        f -> vet[f -> fim] = elem;
        f -> tam++;
    }
}

int fila_vazia(FilaCirc * f) {
    /* Retorna verdadeiro se a Fila estiver vazia */
    return (f -> tam == 0);
}

char remove_fila(FilaCirc * f) {
    if (fila_vazia(f)) {
        /* Verifica se a Fila est vazia */
        printf(Fila vazia\ n);
    } else {
        /* Caso a Fila contenha elemento,  removido o primeiro */
        f -> ini = (f -> ini % (N - 1)) + 1;
        f -> tam--;
    }
}
