Lista * retira(Lista * l, int v) {
    Lista * ant = NULL;
    Lista * p = l;
    while (p != NULL && p -> info != v) {
        ant = p;
        p = p -> prox;
    }
    if (p == NULL)
        return l;
    if (ant == NULL) {
        l = p -> prox;
    } else {
        p -> ant -> prox = p -> prox;
    }
    if (p -> prox != NULL)
        p -> prox -> ant = p -> ant;
    return l;
}

Ordenar a lista duplamente ligada

void Ordena(Lista * l) {
    Lista * p;
    Lista * aux;
    int temp;
    for (p = l; p != NULL; p = p -> prox) {
        for (aux = p -> prox; aux != NULL; aux = aux -> prox) {
            if ((p -> info) > (aux -> info)) {
                temp = p -> info;
                p -> info = aux -> info;
                aux -> info = temp;
            }
        }
    }
}
